class index{
   public static void main(String args[]){

      Scanner sc = new Scanner(System.in);

      System.out.print("Enter number n1:");
      int n1 = sc.nextInt();

      System.out.print("Enter number n2:");
      int n2 = sc.nextInt();

   }
